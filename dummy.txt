[EMITIDARECIBIDA] ,
[PAIS] ,

[GDESCOMPANYNAME] ,
[INVOICEID] ,
[INVOICEDATE] ,
[DEVENGODATE] ,
[TIPOFACTURA] ,
[CLAVEREGIMEN] ,
[CAUSAEXENTA] ,
[TIPONOEXENTA] ,

[BASEIMPONIBLETOTAL] ,

[TIPONOEXENTA] ,

[RECTIFICATIVAINVOICEID] ,
[FECHARECTIFICADA] ,

[CUOTARECTIFICADA] ,
